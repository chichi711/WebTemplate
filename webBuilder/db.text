CREATE TABLE `account` (
  `aID` tinyint(2) UNSIGNED ZEROFILL NOT NULL Primary key auto_increment,
  `mID` tinyint(2) UNSIGNED ZEROFILL NOT NULL ,
  `tID` tinyint(2) UNSIGNED ZEROFILL NOT NULL ,
  `aName` varchar(50) COLLATE utf8_unicode_ci default 'eeWeb template',
  `explanation` varchar(200) COLLATE utf8_unicode_ci,
  `pic` MEDIUMBLOB,
  `date` varchar(20) COLLATE utf8_unicode_ci NOT NULL,
  FOREIGN KEY (mID) REFERENCES member(mID),
  FOREIGN KEY (tID) REFERENCES template(tID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;



CREATE TABLE `pages` (
  `pID` int(2) UNSIGNED ZEROFILL NOT NULL Primary key auto_increment,
  `pName` varchar(20) COLLATE utf8_unicode_ci default 'eeWeb template',
  `aID` tinyint(2) UNSIGNED ZEROFILL NOT NULL,
  `body` varchar(10000) COLLATE utf8_unicode_ci NOT NULL,
  FOREIGN KEY (aID) REFERENCES account(aID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
